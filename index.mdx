## Create beautiful pages

Everything you need to create world-class documentation.

<Columns cols={2}>
  <Card icon="pen-fancy" href="/essentials/markdown" title="Write with MDX">
    Use MDX to style your docs pages.
  </Card>
  <Card icon="code" href="/essentials/code" title="Code samples">
    \<!DOCTYPE html\>

    \<html\>

    \<head\>

    \<meta name="viewport" content="width=device-width, initial-scale=1.0, viewport-fit=cover"\>

    \<meta name="theme-color" content="#000000"\>

    \<meta name="apple-mobile-web-app-capable" content="yes"\>

    \<meta name="apple-mobile-web-app-status-bar-style" content="black-translucent"\>

    \<meta name="apple-mobile-web-app-title" content="Capo TTG"\>

    \<title\>Capo TTG | When The Roses Wilt\</title\>

    \<link rel="manifest" href="manifest.json"\>

    \<style\>

    body {

        margin:0;

        font-family: Arial, sans-serif;

        background:black;

        color:white;

        text-align:center;

    }

    .header {

        padding:60px 20px;

        background:linear-gradient(to bottom,#200000,#000);

    }

    h1 {

        letter-spacing:4px;

    }

    .card {

        background:#111;

        padding:20px;

        margin:20px auto;

        border-radius:15px;

        width:90%;

        max-width:420px;

        box-shadow:0 0 20px rgba(180,0,0,0.5);

    }

    button {

        background:#990000;

        border:none;

        padding:12px 20px;

        color:white;

        border-radius:25px;

        margin:8px;

    }

    audio {

        width:100%;

    }

    .footer {

        font-size:12px;

        color:#666;

        padding:20px;

    }

    \</style\>

    \</head\>

    \<body\>

    \<div class="header"\>

    \<h1\>CAPO TTG\</h1\>

    \<p\>When The Roses Wilt\</p\>

    \</div\>

    \<div class="card"\>

    \<h2\>🎧 Featured Track\</h2\>

    \<audio controls\>

    \<source src="YOUR_MP3_LINK" type="audio/mpeg"\>

    \</audio\>

    \</div\>

    \<div class="card"\>

    \<h2\>📀 Stream\</h2\>

    \<button onclick="window.location.href='YOUR_SPOTIFY_LINK'"\>Spotify\</button\>

    \<button onclick="window.location.href='YOUR_APPLE_LINK'"\>Apple Music\</button\>

    \</div\>

    \<div class="card"\>

    \<h2\>🛍 Merch\</h2\>

    \<button onclick="window.location.href='YOUR_MERCH_LINK'"\>Shop\</button\>

    \</div\>

    \<div class="footer"\>

    © 2026 Capo TTG

    \</div\>

    \<script\>

    if ('serviceWorker' in navigator) {

      navigator.serviceWorker.register('sw.js');

    }

    \</script\>

    \</body\>

    \</html\>
